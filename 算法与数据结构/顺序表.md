* 顺序表就是一个长度可变的数组（动态数组）：使用前先初始化分配空间，长度不够了要扩容，实现增删改查等等功能才是一个数据结构的基本功能，顺序表空间浪费太大，而且不容易插入新元素。
```c
struct User{
	char name[16];
	char address[16];
	char phone[16];
};
typedef struct List{
	struct User *users;
	int capacity;
	int length;
}List;
List * init(int capacity);//初始化动态数组
void expand(List *list);//扩容动态数组
void addData(List *list,char name[16],char phone[16],char address[16]);//添加内容，满了就调用扩容追加
void delData(List *list,char getname[16]);//删除对应名字的所有数据
void updataData(List *list,char getoldname[16],char getnewname[16],char getnewaddress[16],char getnewphone[16]);//更新数据
int searchName(List *list,char getname[16]);//查找所在的地址并返回下标
void printfList(List *list);//打印所有数据
void insertData(List *list,char getoldname[16],char getnewname[16],char getnewaddress[16],char getnewphone[16]);//插入数据
int main(){
	List *l = init(2);
	addData(l,"王一","+86-123456789","192.168.1.1");
	addData(l,"王二","+86-123456789","192.168.1.1");
	addData(l,"王三","+86-123456789","192.168.1.1");
	addData(l,"王四","+86-123456789","192.168.1.1");
	addData(l,"王五","+86-123456789","192.168.1.1");
	addData(l,"王六","+86-123456789","192.168.1.1");
	addData(l,"王七","+86-123456789","192.168.1.1");
	addData(l,"王八","+86-123456789","192.168.1.1");
	addData(l,"王九","+86-123456789","192.168.1.1");
	delData(l,"王三");
	updataData(l,"王五","新王五","255.255.255.0","+86987654321");
	insertData(l,"王七","插入测试","插入测试","插入测试");
	printfList(l);
	printf("长度：%d 容量：%d\n",l->length,l->capacity);
	return 0;
}
List * init(int capacity){	
	List *list=malloc(sizeof(List));//为动态数组分配空间
	list->users=calloc(capacity,sizeof(struct User));//为动态数组中的users数组分配空间
	memset(list->users,0,capacity*sizeof(struct User));//初始化users数组
	list->capacity=capacity;//容量初始化为指定的值
	list->length=0;//长度初始化
	return list;
}
void expand(List *list){
	list->users = realloc(list->users,list->capacity*2*sizeof(struct User));//realloc重新分配空间
	memset(list->users+list->capacity,0,sizeof(struct User)*list->capacity);//初始化扩容后的空间
	list->capacity*=2;//扩容
}
void addData(List *list,char getname[16],char getphone[16],char getaddress[16]){
	if(list->length==list->capacity){
		expand(list);
	}
	for(int i=0;i<list->capacity;i++){
		if(strcmp(list->users[i].name,"\0")==0){
			strcpy(list->users[i].name,getname);
			strcpy(list->users[i].address,getaddress);
			strcpy(list->users[i].phone,getphone);
			list->length++;
			break;
		}
	}
}
void delData(List *list,char getname[16]){
	if(list->length==0||strlen(getname)==0){
		return;
	}
	for(int i=0;i<list->capacity;i++){
		if (strcmp(list->users[i].name, getname) == 0) {
            strcpy(list->users[i].name, "\0");
            strcpy(list->users[i].address, "\0");
            strcpy(list->users[i].phone, "\0");
            list->length--; 
        }
	}
}
int searchName(List *list,char getname[16]){
	if(list->length==0||strcmp(getname,"\0")==0){
		return -1;
	}
	for(int i=0;i<list->capacity;i++){
		if(strcmp(list->users[i].name,getname)==0){
			return i;
		}
	}
}
void updataData(List *list,char getoldname[16],char getnewname[16],char getnewaddress[16],char getnewphone[16]){
	if(list->length==0){
		return;
	}
	for(int i=0;i<list->capacity;i++){
		if(strcmp(list->users[i].name,getoldname)==0){
			strcpy(list->users[i].name,getnewname);
			strcpy(list->users[i].address,getnewaddress);
			strcpy(list->users[i].phone,getnewphone);
			break;
		}
	}
}
void printfList(List *list){
	for(int i=0;i<list->capacity;i++){
		if(strcmp(list->users[i].name,"\0")!=0){
			printf("%s %s %s \n",list->users[i].name,list->users[i].address,list->users[i].phone);
		}
	}
}
void insertData(List *list,char getindexname[16],char getnewname[16],char getnewaddress[16],char getnewphone[16]){
	int index=searchName(list,getindexname);
	if(index <0){
		return;
	}
	if(list->length==list->capacity){
		expand(list);
	}
	int key = index;
	while(strcmp(list->users[key].name,"")!=0){
		key++;
		
	}
	for(int i=key;i>index;i--){
		strcpy(list->users[i].name,list->users[i-1].name);
		strcpy(list->users[i].address,list->users[i-1].address);
		strcpy(list->users[i].phone,list->users[i-1].phone);
	}
	strcpy(list->users[index].name,getnewname);
	strcpy(list->users[index].address,getnewaddress);
	strcpy(list->users[index].phone,getnewphone);
	list->length++;
}
```
